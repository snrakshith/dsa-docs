### 48. Sum of Prime Factors

```javascript
function sumOfPrimeFactors(number) {
  const factors = [];
  for (let i = 2; i <= number; i++) {
    while (number % i === 0) {
      factors.push(i);
      number /= i;
    }
  }
  return [...new Set(factors)].reduce((sum, factor) => sum + factor, 0);
}

console.log(sumOfPrimeFactors(12)); // 5
```
